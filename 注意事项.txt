2、函数可变参数使用 ... 
function average(...)
    result = 0
    local arg = {...} --> arg 为一个表，局部变量,将传入的表保存
    for i, v in ipairs(arg) do
        result = result + v
    end
    print("总共传入 " .. #arg .. " 个数")
    return result / #arg
end

print("平均值为", average(10, 5, 3, 4, 5, 6))
